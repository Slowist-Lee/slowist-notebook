    代码要求如下： 浙大某年度期末考试的评分标准是这样的：

    1. 总评成绩 = 考勤 + 作业x20% + 阶段测试x25% + 理论考试x35% + 实验考试调整分数
    2. 作业、阶段测试、理论考试是百分制。考勤满分5分。
    3. 实验考试共有两道题，第一题10分，第二题15分。
    4. 实验考试和理论考试为上机考试，由机器判卷，给分必然是整数。作业与阶段测试的给分可能有小数。
    5. 只要参加了实验考试，哪怕交了白卷，“实验考试调整分数”就是实验考试两道题总分的开方加10。如果缺考，“实验考试调整分数”为0。
    6. 如果实验考试两道题都未得到全部分数，有一次补考的机会。补考也是两道题，第一题10分，第二题15分。
    7. 如果实验考试补考的两道题也都没能得到全部分数，最终成绩为总评成绩和59的最小值。实验考试调整分数仍按正式考试分数计算。也就是说，实验考试至少需要把一道题全部作对，最终成绩才能及格。
    8. 如果因为实验考试不及格参加了实验考试补考，并且其中至少一道题得到了全部分数，则“实验考试调整分数”强制固定为9，无视答题得分。
    9. 如果理论考试未得到50分或以上，最终成绩会变成总评成绩和59的最小值。也就是说，无论其它项目多高，只要理论考试没有拿到一半的分数，也会被算作不及格。
    10. 最终成绩四舍五入到整数，60分或以上为及格。

    现在，给出一些学生的各部分成绩，请计算各自的最终成绩。

    输入格式:¶

    一行表头，其后是若干行，每行依次是空格隔开的：

    姓名 学号 考勤成绩 作业成绩 阶段测试成绩 理论考试成绩 实验第一题成绩 实验第二题成绩 实验补考第一题成绩 实验补考第二题成绩

    姓名和学号为不超过100字节的字符串。所有成绩可能是数字，也可能是汉字。汉字表示的成绩（比如“缺考”）一律视为0分，不会超过100字节。学生总数不超过10人。

    注意：实验考试及格的同学补考缺考是正常现象。

    输出格式:¶

    每个学生一行，依次是空格隔开的：

    姓名 学号 最终成绩

    #include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <stdbool.h>
#include <math.h>
#include <ctype.h>

// 判断字符串是否为数字
bool is_number(char *str) {
    while (*str) {
        if (!isdigit(*str)) {
            return false;
        }
        str++;
    }
    return true;
}

bool judge_pr_failed(char *a, char *b) {
    int a1;
    int b1;
    char *endptr;
    if (is_number(a)) {
        a1 = strtol(a, &endptr, 10);
    } else {
        a1 = 0;
    }
    if (is_number(b)) {
        b1 = strtol(b, &endptr, 10);
    } else {
        b1 = 0;
    }
    if (a1 < 10 && b1 < 15) { // 使用整数常量进行比较
        return false;
    } else {
        return true;
    }
}

int min(int a, int b) {
    if (a < b) {
        return a;
    } else {
        return b;
    }
}

int main(void) {
    char *lst[10000];
    int i = 0;

    while (i < 10000) {
        char str[256]; // 假设每行最多256个字符
        if (fgets(str, sizeof(str), stdin) == NULL) {
            break;
        }

        // 为当前行分配内存并存储
        lst[i] = malloc(strlen(str) + 1);
        strcpy(lst[i], str);
        i++;
    }

    char *name[i];
    char *id[i];
    char *attendence[i];
    char *homework[i];
    char *quiz[i];
    char *th_test[i];
    char *pr_test1[i];
    char *pr_test2[i];
    char *pr_atest1[i];
    char *pr_atest2[i];
    char *endptr;
    bool pass = true;
    for (int j = 1; j < i; j++) {
        char *rest = lst[j];
        name[j] = strtok(rest, " "); // 获取姓名
        id[j] = strtok(NULL, " "); // 获取学号 
        attendence[j] = strtok(NULL, " "); // 考勤成绩 
        homework[j] = strtok(NULL, " "); // 作业成绩 
        quiz[j] = strtok(NULL, " "); // 小测成绩
        th_test[j] = strtok(NULL, " "); // 理论考试成绩
        pr_test1[j] = strtok(NULL, " "); // 实验考试成绩 1
        pr_test2[j] = strtok(NULL, " "); // 实验考试成绩 2
        pr_atest1[j] = strtok(NULL, " "); // 实验补考成绩 1
        pr_atest2[j] = strtok(NULL, " "); // 实验补考成绩 2
    }
    double grade[i];
    int n = 0;
    double Att;
    double Hom;
    double Quiz;
    double Th_test;
    double Pr_test = 0.0;
    for (n = 1; n < i; n++) {
        Att = strtod(attendence[n], &endptr);
        grade[n] = Att;

        Hom = strtod(homework[n], &endptr);
        grade[n] += 0.2 * Hom;

        if (is_number(quiz[n])) {
            Quiz = strtod(quiz[n], &endptr);
            grade[n] += 0.25 * Quiz;
        }



        if (judge_pr_failed(pr_test1[n], pr_test2[n])) {
            pass = true;
            Pr_test += strtod(pr_test1[n], &endptr);
            Pr_test += strtod(pr_test2[n], &endptr);
            Pr_test = 10.0 + sqrt(Pr_test);
            grade[n] += Pr_test; // 第一次就过了，正常计算实验得分
        } else {
            // 第一次没过，需要判断第二次有没有过
            if (judge_pr_failed(pr_atest1[n], pr_atest2[n])) {
                pass = true;
                grade[n] += 9; // 第二次过了，固定是9分
            } else {
                pass = false; // 直接挂科
                if (is_number(pr_test1[n])) {
                    Pr_test += strtod(pr_test1[n], &endptr);
                }
                if (is_number(pr_test2[n])) {
                    Pr_test += strtod(pr_test2[n], &endptr);
                }
                Pr_test = 10.0 + sqrt(Pr_test);
                grade[n] += Pr_test; // 也照样计算实验得分
            }

        }

        Th_test = strtod(th_test[n], &endptr);
        grade[n] += 0.35 * Th_test;
        if (Th_test < 60) {
            pass = false;
        }
        if (pass == false) {
             grade[n] = min(59, (int)grade[n]);
        }

    }

    // 输出
    for (int x = 1; x < i-1; x++) {
        printf("%s %s %.0f\n", name[x], id[x], round(grade[x]));
    }
    printf("%s %s %.0f", name[i-1], id[i-1], round(grade[i-1]));
    // 释放动态分配的内存
    for (int s = 0; s < i; s++) {
        free(lst[s]);
    }

    return 0;
}

        Att = strtod(attendence[n], &endptr);
        grade[n] = Att;

        Hom = strtod(homework[n], &endptr);
        grade[n] += 0.2 * Hom;

        if (is_number(quiz[n])) {
            Quiz = strtod(quiz[n], &endptr);
            grade[n] += 0.25 * Quiz;
        }

        // 添加调试信息
        printf("Student: %s, ID: %s\n", name[n], id[n]);
        printf("Initial grade: %.2f\n", grade[n]);

        Pr_test = 0.0; //4 nonsense
        pass = true;
        if (judge_pr_failed(pr_test1[n], pr_test2[n])) {
            Pr_test += strtod(pr_test1[n], &endptr);
            Pr_test += strtod(pr_test2[n], &endptr);
            Pr_test = 10.0 + sqrt(Pr_test);
            grade[n] += Pr_test; // 第一次就过了，正常计算实验得分
            printf("Passed first experiment, Pr_test: %.2f\n", Pr_test);
        } else {
            // 第一次没过，需要判断第二次有没有过
            if (judge_pr_failed(pr_atest1[n], pr_atest2[n])) {
                grade[n] += 9; // 第二次过了，固定是9分
                printf("Passed second experiment, fixed Pr_test: 9\n");
            } else {
                pass = false; // 直接挂科
                if (is_number(pr_test1[n])) {
                    Pr_test += strtod(pr_test1[n], &endptr);
                }
                if (is_number(pr_test2[n])) {
                    Pr_test += strtod(pr_test2[n], &endptr);
                }
                Pr_test = 10.0 + sqrt(Pr_test);
                grade[n] += Pr_test; // 也照样计算实验得分
                printf("Failed both experiments, Pr_test: %.2f\n", Pr_test);
            }
        }

        Th_test = strtod(th_test[n], &endptr);
        grade[n] += 0.35 * Th_test;
        printf("Theory test: %.2f, Total grade so far: %.2f\n", Th_test, grade[n]);

        if (Th_test < 50) {
            pass = false;
            printf("Theory test failed, score < 50\n");
        }
        if (pass == false) {
            grade[n] = min(59, (int)grade[n]);
            printf("Final grade adjusted to minimum of 59 or current total\n");
        }

    }

    // 输出
    for (int x = 1; x < i-1; x++) {
        printf("%s %s %.0f\n", name[x], id[x], round(grade[x]));
    }
    printf("%s %s %.0f", name[i-1], id[i-1], round(grade[i-1]));
    // 释放动态分配的内存
    for (int s = 0; s < i; s++) {
        free(lst[s]);
    }

    return 0;
}
