    import re

def convert_ubb_to_markdown_link(s):
    # 使用正则表达式匹配UBB链接格式
    pattern = r'\[url=(.*?)\](.*?)\[\/url\]'
    # 替换为Markdown链接格式
    return re.sub(pattern, r'[\2](\1)', s)

# 示例UBB链接
ubb_link = '[url=/topic/5378610#1]>>查看原帖<<[/url]'
# 转换为Markdown链接
markdown_link = convert_ubb_to_markdown_link(ubb_link)

print(markdown_link)  # 输出: [查看原帖](https://www.cc98.org/topic/5378610#1)

    这段代码定义了一个convert_ubb_to_markdown_link函数，它接受一个字符串参数s，这个参数是一个UBB格式的链接。函数使用正则表达式来查找匹配UBB链接的模式，然后使用re.sub()函数将找到的UBB链接替换为Markdown格式的链接。

    请注意，上述代码中的正则表达式r'\[url=(.*?)\](.*?)\[\/url\]'假定UBB链接格式是严格遵循[url=...][...][/url]的结构。如果UBB链接的格式有所不同，你可能需要调整正则表达式以适应不同的情况。

    此外，示例中的ubb_link变量需要根据实际情况替换为包含实际链接的字符串。在实际使用中，你可能需要将这个函数应用到包含多个UBB链接的整个文本上。

    for tag in tags:
        start_tag, end_tag = tags[tag]['start'], tags[tag]['end']
        # 检查是否需要特殊处理，例如链接转换
        if tag == 'url':
            # 使用正则表达式替换UBB链接
            s = re.sub(r'\[url=(.*?)\](.*?)\[\/url\]', r'[\2](https://www.cc98.org/\1)', s)
        else:
            # 对于其他标签，使用之前的替换逻辑

                s = s[:start_index] + end_tag + s[end_index:]

        function convertToUBB() {
            // 获取输入的Markdown内容
            let markdownText = document.getElementById('inputArea').value;

            // 定义转换规则
            markdownText = markdownText.replace(/\*\*(.*?)\*\*/g, '[b]$1[/b]');
            markdownText = markdownText.replace(/<u>(.*?)<\/u>/g, '[u]$1[/u]');
            markdownText = markdownText.replace(/~~(.*?)~~/g, '[del]$1[/del]');
            markdownText = markdownText.replace(/<font color="#0000FF">(.*?)<\/font>/g, '[color=blue]$1[/color]');
            markdownText = markdownText.replace(/^> (.*)$/gm, '[quote]$1[/quote]');

            // 处理url标签
            markdownText = markdownText.replace(/\[查看原帖\]\(https:\/\/www\.cc98\.org\/(.*?)\)/g, '[url=$1]>>查看原帖<<[/url]');

            // 显示转换后的结果
            document.getElementById('outputArea').innerText = markdownText;
